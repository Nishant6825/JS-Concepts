let str = '()';


function isvalid(s) {
    let stack = [];
    for (const element of s) {
        if (element === '(') stack.push(')')
        else if (element === '{') stack.push('}')
        else if (element === '[') stack.push(']')
        else if (element !== stack.pop()) return false
    }
    return stack.length === 0;
}

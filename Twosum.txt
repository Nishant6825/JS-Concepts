let arr = [2, 7, 11, 15, 17, 1]


function GetTarget(arr, target) {
    let elements = []

    for (let i = 0; i < arr.length; i++) {
        for (let j = i + 1; j < arr.length; j++) {
            if (arr[i] + arr[j] === target) {
                elements.push([arr[i], arr[j]]);
                // return [arr[i], arr[j]]
            }
        }
    }
    return elements;
}

console.log(GetTarget(arr, 18));

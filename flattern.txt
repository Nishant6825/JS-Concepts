function falt(arr) {
    let result = [];

    for (let i = 0; i < arr.length; i++) {
        if (Array.isArray(arr[i])) {
            result.push(...falt(arr[i]));
        } else {
            result.push(arr[i]);
        }

    }
    return result;
}


console.log(falt([1, [2, [4, [5, 6,], 9], 9]]));
